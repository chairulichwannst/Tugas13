    private void btnhasilActionPerformed(java.awt.event.ActionEvent evt) {                                         
        switch (pilih){
            case 1:
                bil2 = Double.parseDouble(angka);   
                jumlah = bil1+bil2; 
                angka = Double.toString(jumlah); 
                tampil.setText(angka);
                break; 
            case 2:
                bil2 = Double.parseDouble(angka);   
                jumlah = bil1-bil2; 
                angka = Double.toString(jumlah); 
                tampil.setText(angka);
                break; 
            case 3:
                bil2 = Double.parseDouble(angka);   
                jumlah = bil1*bil2; 
                angka = Double.toString(jumlah); 
                tampil.setText(angka);
                break; 
            case 4:
                bil2 = Double.parseDouble(angka);   
                jumlah = bil1/bil2; 
                angka = Double.toString(jumlah); 
                tampil.setText(angka);
                break; 
            default: 
                break; 
        }
    }
